// Number (숫자) 타입
let studentAge = 20;
let height = 175.5;
let temperature = -5;

console.log(studentAge); // 결과: 20
console.log(typeof studentAge); // 결과: "number"


// String (문자열) 타입
let studentName = "김개발";

console.log(studentName); // 결과: 김개발
console.log(typeof studentName); // 결과: "string"

/*------------------------------------------------------------------------------*/ // 타입별 처리 함수
function processUserData(user) {
console.log("=== 사용자 정보 분석 ===");
// 문자열 처리
if (typeof user.name === "string") {
console.log("이름: " + user.name + " (길이: " + user.name.length + "자)");
}

// 숫자 처리
if (typeof user.age === "number") {
console.log("나이: " + user.age + "세 (성인 여부: " + (user.age >= 20) + ")");
}

// 불린 처리
if (typeof user.isStudent === "boolean") {
console.log("학생 여부: " + (user.isStudent ? "재학중" : "비학생"));
}

// null과 undefined 처리
if (user.graduationYear === null) {
console.log("졸업년도: 미정 (재학중)");
}

if (user.favoriteColor === undefined) {
console.log("선호색상: 아직 선택하지 않음");
}
}
processUserData(userInfo);
